using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace OLTHDT
{
    class PhongBan
    {
      private  string mapb;
      private string tenphong;

        public string Mapb
      {
          get { return mapb; }
          set { mapb = value; }
      }
        public string Tenphong
        {
            get { return tenphong; }
            set { tenphong = value; }
        }
        public PhongBan()
        {
            mapb = "";
            tenphong = "";
         }

        public PhongBan(string ma,string tenp)
        {
            mapb = ma;
            tenphong = tenp;
        }


    }
}
